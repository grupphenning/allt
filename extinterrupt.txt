allt detta i main:

//aktivera interrupt på INT0 och INT1
setbit(EIMSK, INT0);
setbit(EIMSK, INT1);

//aktivera interrupt-request på "any change"
setbit(EICRA, ISC00);
//aktivera global interrupts
sei();
